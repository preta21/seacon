Verilog实现快速傅里叶变换

  • 2026-06-03

module butterfly_2point #(
    parameter DATA_WIDTH = 16 // 输入数据位宽（包含符号位）
)(
    input  wire                      clk,      // 时钟信号
    input  wire                      rst_n,    // 异步复位信号（低电平有效）
    
    // 输入复数 A (Real & Imag)
    input  wire signed [DATA_WIDTH-1:0] in_A_re,
    input  wire signed [DATA_WIDTH-1:0] in_A_im,
    
    // 输入复数 B (Real & Imag)
    input  wire signed [DATA_WIDTH-1:0] in_B_re,
    input  wire signed [DATA_WIDTH-1:0] in_B_im,
    
    // 输出复数 X0 = A + B (位宽加1防止溢出)
    output reg  signed [DATA_WIDTH:0]   out_X0_re,
    output reg  signed [DATA_WIDTH:0]   out_X0_im,
    
    // 输出复数 X1 = A - B (位宽加1防止溢出)
    output reg  signed [DATA_WIDTH:0]   out_X1_re,
    output reg  signed [DATA_WIDTH:0]   out_X1_im
);


    // 时序逻辑：在时钟上升沿锁存计算结果，提高硬件的时序性能
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            out_X0_re <= 'd0;
            out_X0_im <= 'd0;
            out_X1_re <= 'd0;
            out_X1_im <= 'd0;
        end else begin
            // X0 = A + B
            out_X0_re <= in_A_re + in_B_re;
            out_X0_im <= in_A_im + in_B_im;
            
            // X1 = A - B
            out_X1_re <= in_A_re - in_B_re;
            out_X1_im <= in_A_im - in_B_im;
        end
    end


endmodule
